class Solution:
    def search(self, nums: List[int], target: int) -> int:
        length = len(nums) - 1
        right_most = nums[length-1]
        if len(nums) == 1:
            if nums[0] == target:
                return 0
            else:
                return -1
        
        mid = length // 2
        pivot = 0
        left = 0
        right = length
        while left < right:
            mid = (left + right) // 2
            if nums[mid] > nums[right]:
                left = mid + 1
            else:
                right = mid
        
        pivot = left

        left = 0 
        right = length

        if target >= nums[pivot] and target <= nums[length]:
            left = pivot
        else:
            right = pivot -1

        
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                return mid
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
        
        return -1
