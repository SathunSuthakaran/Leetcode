class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        n = len(intervals)
        intervals = sorted(intervals, key=lambda x: x[1])
        temp_end = intervals[0][1]
        count = 0
        for i in range(1, n):
            if intervals[i][0] < temp_end:
                count += 1
            else:
                temp_end = intervals[i][1]
        return count
