class Solution:
    def mergeTriplets(self, triplets: List[List[int]], target: List[int]) -> bool:
        best = [0, 0, 0]
        for x,y,z in triplets:
            if x <= target[0] and y <= target[1] and z <= target[2]:
                best[0] = max(x, best[0])
                best[1] = max(y, best[1])
                best[2] = max(z, best[2])
        return best == target
