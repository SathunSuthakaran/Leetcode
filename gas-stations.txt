class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:
        ret = []
        curr_tank = 0
        total_tank  = 0
        start = 0
        for i in range(len(gas)):
            curr_tank += (gas[i]-cost[i])
            total_tank += gas[i] - cost[i]
            if curr_tank < 0:
                start = i + 1
                curr_tank = 0
        
        if total_tank >= 0:
            return start
        else:
            return -1
