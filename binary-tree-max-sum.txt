# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def maxPathSum(self, root: Optional[TreeNode]) -> int:
        paths = []
        def recurse(node):
            if not node:
                return 0
            L = recurse(node.left)
            R = recurse(node.right)
            through = node.val + max(0, L) + max(0, R)
            paths.append(through)
            return node.val + max(0, L, R)
        recurse(root)
        return(max(paths))
